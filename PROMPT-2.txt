import re
import logging
from django.shortcuts import get_object_or_404, redirect, render
from django.contrib import messages
from django.contrib.auth.models import User
from django.core.mail import send_mail

from django.db import transaction

logger = logging.getLogger(__name__)

# ----------------------------------
# APPROVE UNIVERSITY
# ----------------------------------
def admin_approve(request, pk):
    """
    Approve a university request.
    Password format: Smartedu_<cleaned_collegeID>_<cleaned_PIN>
    """

    uni = get_object_or_404(UniversityRequest, pk=pk)

    if uni.status != "Pending":
        messages.warning(request, "Only pending requests can be approved.")
        return redirect("university_reg_show")

    # -------------------------
    # PASSWORD LOGIC (UPDATED)
    # -------------------------
    # Remove special characters from college_id and pincode
    special_chars_pattern = r"[,!_\-=*&^%$#@]"
    clean_college_id = re.sub(special_chars_pattern, "", str(uni.college_id))
    clean_pincode = re.sub(special_chars_pattern, "", str(uni.pincode))

    raw_password = f"Smartedu_{clean_college_id}_{clean_pincode}"

    # Clean name for email greeting
    email_name = re.sub(r"\d+", "", uni.university_email.split("@")[0]).capitalize()

    context = {
        "clean_name": email_name,
        "college_id": uni.college_id,
        "email": uni.university_email,
        "password": raw_password,
        "login_url": request.build_absolute_uri("/login/"),
    }

    try:
        with transaction.atomic():

            # -------------------------
            # DUPLICATE CHECKS
            # -------------------------
            if ApprovedUniversityAccount.objects.filter(college_id=uni.college_id).exists():
                messages.error(
                    request,
                    format_html(
                        'College ID already exists. '
                        'Please use <a href="{}" target="_blank">Admin Panel</a> to make changes.',
                        ADMIN_URL
                    )
                )
                return redirect("university_reg_show")

            if User.objects.filter(username=uni.college_id).exists():
                messages.error(
                    request,
                    format_html(
                        'User already exists. '
                        'Please use <a href="{}" target="_blank">Admin Panel</a> to make changes.',
                        ADMIN_URL
                    )
                )
                return redirect("university_reg_show")

            # -------------------------
            # SEND EMAIL
            # -------------------------
            html_message = render_to_string("emails/approval_email.html", context)
            send_mail(
                subject="SmartEduCare – University Account Approved",
                message=strip_tags(html_message),
                from_email=settings.DEFAULT_FROM_EMAIL,
                recipient_list=[uni.university_email],
                html_message=html_message,
                fail_silently=False,
            )

            # -------------------------
            # CREATE AUTH USER
            # -------------------------
            user = User.objects.create_user(
                username=uni.college_id,
                email=uni.university_email,
                password=raw_password,
                first_name=uni.name[:30],
            )

            # -------------------------
            # CREATE APPROVED UNIVERSITY ACCOUNT
            # -------------------------
            ApprovedUniversityAccount.objects.create(
                user=user,
                college_id=uni.college_id,
                name=uni.name,
                university_email=uni.university_email,
                country=uni.country,
                state=uni.state,
                pincode=uni.pincode,
                contact_number=uni.contact_number,
                website=uni.website,
            )

            # -------------------------
            # SAVE LOGIN CREDENTIALS
            # -------------------------
            ApprovedUniversityCredential.objects.create(
                college_id=uni.college_id,
                email=uni.university_email,
                password=raw_password,
            )

            # -------------------------
            # UPDATE STATUS
            # -------------------------
            uni.status = "Approved"
            uni.save(update_fields=["status"])

            messages.success(
                request,
                "University approved successfully. Credentials emailed and saved."
            )

    except Exception as e:
        logger.exception("Approval failed")
        messages.error(request, f"Approval failed: {e}")

    return redirect("university_reg_show")

from django.shortcuts import get_object_or_404, redirect
from django.contrib import messages
from django.core.mail import send_mail
from django.template.loader import render_to_string
from django.utils.html import strip_tags
from university_app.models import UniversityRequest

def send_correction_email(request, pk):
    uni = get_object_or_404(UniversityRequest, pk=pk)
    
    if uni.status != "Pending":
        messages.warning(request, "Only pending requests can receive correction emails.")
        return redirect("university_reg_show")

    remark = request.GET.get("remark", "").strip()
    
    if not remark:
        messages.error(request, "Remarks are required to send correction email.")
        return redirect("university_reg_show")

    try:
        html_message = render_to_string("emails/correction_email.html", {
            "uni": uni,
            "remark": remark,
        })

        send_mail(
            subject="SmartEduCare – Correction Required",
            message=strip_tags(html_message),
            from_email=settings.DEFAULT_FROM_EMAIL,   # ← important: use this instead of None
            recipient_list=[uni.university_email],
            html_message=html_message,
            fail_silently=False,
        )

        # Mark rejected and delete inside transaction
        with transaction.atomic():
            uni.status = "Rejected"
            uni.save(update_fields=["status"])
            uni.delete()

        messages.success(
            request,
            f"Correction email sent to {uni.university_email}. Request marked Rejected and removed."
        )

    except Exception as e:
        logger.exception("Correction email failed")
        messages.error(request, f"Failed to send correction email: {str(e)}")

    return redirect("university_reg_show")
